\section{기본 표기}
선형대수에서 행렬 $A \in \R^{m\times n}$의 전치는 $A^{\mathsf T}$로 쓰고,
프로베니우스 노름은 $\norm{A}_F = \sqrt{\sum_{i,j} a_{ij}^2}$로 정의한다.

\begin{equation}
  \nabla \cdot \mathbf{E} = \frac{\rho}{\varepsilon_0},
  \qquad
  \oint_{\partial \Omega} \mathbf{F}\cdot \mathrm{d}\mathbf{S}
    = \iiint_\Omega (\nabla\cdot\mathbf{F})\,\mathrm{d}V.
\end{equation}

\section{정의와 정리}

\begin{definition}[노름 공간]
  벡터공간 $V$ 위의 함수 $\norm{\cdot} : V \to \R$가 (i) $\norm{v}\ge 0$,
  (ii) $\norm{v}=0 \iff v=0$, (iii) $\norm{\alpha v}=\abs{\alpha}\norm{v}$,
  (iv) $\norm{v+w}\le \norm{v}+\norm{w}$를 만족하면 \emph{노름}이라 한다.
\end{definition}

\begin{theorem}[코시--슈바르츠 부등식]
  내적공간 $V$의 모든 $u, v \in V$에 대해
  \[
    \abs{\langle u, v\rangle} \le \norm{u}\,\norm{v}.
  \]
\end{theorem}
